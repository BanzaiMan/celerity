def self.create(host, port)
  # if the connection fails, we won't spend time loading json
  socket = TCPSocket.new(host, port)
  require "json"
  new(socket)
end

def image_data
  send_data('method' => 'image_data')
  data = read_data
  data['image'] || data['error']
end
