def save(filename)
  assert_exists
  image_reader = @object.getImageReader
  file = java.io.File.new(filename)
  buffered_image = image_reader.read(0);
  javax.imageio.ImageIO.write(buffered_image, image_reader.getFormatName(), file)
end
